import java.awt. *;

class SuccessDialog extends Dialog {

   private Button OK;
   private Panel buttonPanel;
   private Label text;
   private Frame thisFrame;
   Font f1 = new Font("SansSerif", Font.BOLD, 18);

   SuccessDialog(Frame f, String title) {
      super(f, title, true);
      thisFrame = f;
      this.resize(320, 125);
      OK = new Button("OK");

      buttonPanel = new Panel();
      buttonPanel.setLayout(new FlowLayout());
      buttonPanel.setBackground(Color.yellow);
      buttonPanel.add(OK);
      add("South", buttonPanel);

      text = new Label("Well done. You found all the mines.", Label.CENTER);
      text.setFont(f1);
      text.setBackground(Color.yellow);
      add("Center", text);
   } 

   public boolean action(Event ev, Object obj) {
      if (ev.target == OK) {
         this.hide();
         this.dispose();
      }
      return true;
   }
}
